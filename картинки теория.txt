для body:
oveflow-y:scroll - прокуртка страницы
background-image: url(image.jpg) - фоновое изображение
background-size: cover - размер фонового изображения
background-position: center - расположение фонового изображения по центру
background-attachment: fixed - фиксированное фоновое изображение
display: flex; flex-direction: column; align-items: center; - выравнивание по центру
background-color: rgba(255, 255, 255) - цвет фона

для content:
background-color: rgba(255, 255, 255, 0.5) - 50% прозрачности
border-radius: 10px - закругление углов
box-shadow: 0 0 5px 0 rgba(0, 0, 0, 0.2) - тень

для создания рамки:
border: 1px solid #000000; - цвет, толщина и тип линии
width: 256px; height: 256px; - размеры
display-flex; justify-content: center; align-items: center; - выравнивание по центру
overflow: hidden; - скрывает содержимое выходящее за границы

для большой картинки:
max-width: 100%; max-height: 100%; - масштабирование картинки
object-fit: contain; - содержимое картинки умещается в блок, сохранение пропорций
